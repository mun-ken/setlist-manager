# Auto-update SSL: PyInstaller-bundles på Windows mangler ofte CA-certifikater
# så ssl.create_default_context() fejler med CERTIFICATE_VERIFY_FAILED.
# certifi giver os en pålidelig CA-bundle som fallback.
certifi>=2024.0.0

# NDI broadcast (til OBS/vMix/ATEM):
# numpy + ndi-python. Begge er optional — appen kører fint uden dem.
# Hvis ndi-python ikke kan installeres (kræver NDI Runtime fra
# https://ndi.video/tools/), springer requirements.txt det over og
# NDI-features bliver gråtonet i UI'et.
numpy>=1.24
ndi-python>=5.1.1.5; platform_system == "Windows" or platform_system == "Darwin"

# Globale hotkeys (Live mode — virker selv hvis OBS har focus):
# keyboard-library bruger et lavt-niveau keyboard hook. På Windows kræver
# det INGEN admin/elevation. På macOS kræver det "Accessibility"-permission
# (brugeren bliver bedt om det første gang). På Linux kræver det typisk root
# og er derfor ikke pålideligt — vi falder pænt tilbage til vindue-focus.
keyboard>=0.13.5; platform_system == "Windows" or platform_system == "Darwin"
